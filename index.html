function doPost(e) {
  try {
    var raw = e.postData.contents;
    var data = JSON.parse(raw);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

    // Add headers if empty
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        "Timestamp","Reg No","Name","B Form","Admission Date",
        "DOB","Class","Gender","Father Name","Father CNIC",
        "Mother CNIC","No. Children","Mobile","Address","Photo"
      ]);
      sheet.getRange(1,1,1,15).setFontWeight("bold")
        .setBackground("#0f1f3d").setFontColor("#ffffff");
    }

    sheet.appendRow([
      new Date().toLocaleString(),
      data.regNo, data.name, data.bForm, data.admissionDate,
      data.dob, data.class, data.gender, data.fatherName,
      data.fatherCnic, data.motherCnic, data.numChildren,
      data.mobile, data.address, data.photoUrl || ""
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({status:"success"}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:"error",message:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
